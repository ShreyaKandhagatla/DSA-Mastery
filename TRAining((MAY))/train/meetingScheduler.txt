i, j = 0, 0
slots1.sort()
slots2.sort()
while i < len(slots1) and j < len(slots2):
    start = max(slots1[i][0], slots2[j][0])
    end = min(slots1[i][1],  slots2[j][1])
    if end - start >= duration:
        return (start, start + duration)
    if slots1[i][1] > slots2[i][1]:
        i += 1
    else:
        j += 1
return (-1, -1)
